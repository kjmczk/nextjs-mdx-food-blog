ingredients:
  [
    '1 tomato, sliced',
    '1/2 spring onion, chopped',
    '1/2 tbsp chopped basil',
    '1.5 tbsp olive oil',
    '1/2 tbsp ponzu sauce',
    '1/2 tbsp soy sauce',
    '1/2 tsp grated garlic',
  ]
directions:
  [
    'Mix the chopped spring onions and the dressing ingredients.',
    'Put the dressing on the sliced tomatoes.',
    'Put the chopped basil on top.',
  ]
